### 定义

区分事物内部的状态

### 使用场景

状态机

### 实现

把事物的每种状态都封装成单独的类，跟此种状态有关的行为都被封装在这个类的内部

```javascript
var FSM = {
  walk: {
    attack: function () {
      console.log('攻击')  
    },
    defense: function () {
      console.log('防御')  
    },
    jump: function () {
      console.log('跳跃')  
    }  
  },
  attack: {
    walk: function () {
      console.log('攻击的时候不能行走')  
    },
    defense: function () {
      console.log('攻击的时候不能防御')  
    },
    jump: function () {
      console.log('攻击的时候不能跳跃')  
    }  
  }  
}
```
